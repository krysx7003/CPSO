def displayEKG(start,end):
    data_frame.columns = ['data']
    new_data = data_frame.iloc[ start :end ].copy()
    plt.figure(figsize=(20,5))
    plt.plot(new_data['data']) #rysowanie wykresu
    plt.grid(True)
    plt.title("EKG")
    plt.show()

Zadanie nr2

Wyznaczono transformant Fouriera korzystajc z biblioteki numpy oraz funkcji fft. Na podstawie transformanty przedstawiono widmo amplitudowe w funkcji czstotliwoci.

Korzystajc z zewntrznych rde wykonano dodatkowe kroki w celu prawidowego wyznaczenia widma amplitudowego:

  • Wyprowadzono amplitud transformanty Fouriera,
  • Znormalizowano zakres amplitudy aby maksymaln wartoci byo 1,
  • Wygenerowao prawidowe czstotliwoci dla zadanego zakresu,
  • Wyznaczono widmo w dodatnim zakresie czstotliwoci,
In[25]:
#Generacja i wywietlenie widma sygnau
fs = 360
t = len(data_frame)

fourier1 = np.fft.fft(data_frame['data']) #transformata fouriera

#dodatkowe kroki
widmo = np.abs(fourier1)
abs_widmo = widmo / np.max(widmo)
freq = np.fft.fftfreq(t,1/fs)
pos_freq = freq[:len(freq)//2] #czstotliwoci w zakresie [0,fs/2]
pos_widmo = abs_widmo[:len(abs_widmo)//2] #widmo amplitudowe (cz dodatnia)
#koniec dodatkowych krokw

plt.figure(figsize=(20,5))
plt.plot(pos_freq,pos_widmo)
plt.xlim(right = fs/2)
plt.grid(True)
plt.title("Widmo amplitudowe")
plt.xlabel('Czstotliwo (Hz)')
plt.ylabel('Amplituda')

plt.show()

#Wizualizacja rnicy odwrotnej transformaty(Utworzonej na podstawie widma) i sygnau
def displayFourier(start,end):
    inv_fourier = np.fft.ifft(fourier1).real
    inv_fourier = data_frame['data'] - inv_fourier
    inv_fourier = inv_fourier.values[start:end]
    plt.figure(figsize=(20,5))
    plt.plot(data_frame['data'][start:end], label="Oryginalny sygna")
    plt.plot(inv_fourier, linestyle="dashed", alpha=0.7, label="Rnica midzy oryginaem a odzyskanym sygnaem")
    plt.grid(True)
    plt.legend()
    plt.title("EKG")
    plt.show()
