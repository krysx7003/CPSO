if len(img.shape)==3:
    img = np.mean(img,axis=2).astype(np.uint8)
mode = input("Podaj pozioma/pionowa:")

if( mode == "pozioma"):
    line_num = int(input("Podaj wysoko: "))
    gray_val = img[line_num,:]

elif(mode == "pionowa"):
    line_num = int(input("Podaj szeroko: "))
    gray_val = img[:,line_num]

else:
    print("Eee")

if(mode == "pozioma"):
    x = np.arange(img.shape[1])
    y = line_num * np.ones(img.shape[1])
elif mode == "pionowa":
    x = line_num * np.ones(img.shape[0])
    y = np.arange(img.shape[0])

plt.subplot(1, 2, 1)
plt.plot(x, y, color='red', linewidth=1)
plt.imshow(img, cmap='gray')
plt.title('Obraz w skali szaroci')
plt.xlabel('Szeroko')
plt.ylabel('Wysoko')
plt.subplot(1, 2, 2)
plt.plot(gray_val)
plt.title('Histogram skali szaroci')
plt.xlabel('Warto pikseli')
plt.ylabel('Czsto')
plt.show()

Na histogramie widoczne s poziomy szaroci pikseli znajdujcych si na wybranej linii obrazu
O X przedstawia pozycj piksela na tej linii, natomiast o Y odpowiada wartoci poziomu szaroci (od 0 do 255).
Analizujc wykres, mona zauway rozkad jasnoci na tej konkretnej linii obrazu, co pozwala oceni kontrast, obecno struktur lub jednolito ta w tym fragmencie zdjcia.
