import matplotlib.pyplot as plt
from PIL import Image 
from skimage import exposure
import skimage.morphology as morp
from skimage.filters import rank
import numpy as np
In[24]:
# Zaadowanie pliku .tiff
img = np.array( Image.open("src/hidden-symbols.tif").convert('L') )
In[25]:
#Wywietlenie zaadowanego obrazu
img_local = rank.equalize(img, morp.disk(6))
img_stat = exposure.equalize_adapthist(img, clip_limit=1.15)

plt.figure()
plt.subplot(2,3,1)
plt.imshow(img, cmap='gray')

plt.subplot(2,3,2)
plt.imshow(img_local, cmap='gray')

plt.subplot(2,3,3)
plt.imshow(img_stat, cmap='gray')

plt.tight_layout()
plt.show()
